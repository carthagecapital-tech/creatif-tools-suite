function getCategory(name) { return CATEGORIES.find(c => c.name === name) || CATEGORIES[0]; }

function calculateReferralFee(categoryName, sellingPrice) {
  const cat = getCategory(categoryName);
  if (!cat) return 0;
  if (cat.tiered) {
    const t1Amount = Math.min(sellingPrice, cat.tier1Threshold || 0);
    const t1 = t1Amount * cat.rate;
    const t2 = Math.max(0, sellingPrice - (cat.tier1Threshold || 0)) * (cat.tier2Rate || 0);
    return Math.max(t1 + t2, cat.minFee || 0);
  }
  return Math.max(sellingPrice * cat.rate, cat.minFee || 0);
}

function getEffectiveReferralRate(categoryName) {
  const cat = getCategory(categoryName);
  if (!cat) return 0;
  return cat.rate;
}

function calculateFBAFee(sizeTier, weightLbs) {
  const tiers = FBA_FEES[sizeTier];
  if (!tiers) return 0;
  const weightOz = weightLbs * 16;
  for (const tier of tiers) {
    if (tier.fee !== null) {
      if (weightOz <= tier.max) return tier.fee;
    } else {
      const tierWeightLbs = weightLbs;
      if (tierWeightLbs <= (tier.max === Infinity ? Infinity : tier.max / 16)) {
        return tier.base + Math.max(0, tierWeightLbs - tier.above) * tier.perLb;
      }
    }
  }
  // Fallback: use last tier's formula
  const last = tiers[tiers.length - 1];
  if (last.fee !== null) return last.fee;
  return last.base + Math.max(0, weightLbs - last.above) * last.perLb;
}

function calculateStorageFee(sizeTier, isQ4, volumeEstimate) {
  const rateType = (sizeTier === 'Standard Small' || sizeTier === 'Standard Large') ? 'standard' : 'oversized';
  const rate = isQ4 ? STORAGE_RATES[rateType].octDec : STORAGE_RATES[rateType].janSep;
  return rate * (volumeEstimate || 0.1);
}

function getBadge(netProfit, margin) {
  if (netProfit < 0) return 'loss';
  if (margin < 20) return 'thin';
  if (margin <= 40) return 'healthy';
  return 'strong';
}

function calculateProduct(p) {
  const referralFee = calculateReferralFee(p.category, p.sellingPrice);
  const fbaFee = calculateFBAFee(p.sizeTier, p.weight);
  const storageFeePerUnit = calculateStorageFee(p.sizeTier, p.storageMonth === 'q4', p.storageVolumeEstimate);
  const totalAmazonFees = referralFee + fbaFee + storageFeePerUnit;
  const totalCostPerUnit = p.productCost + p.prepCost + p.inboundShipping;
  const totalDeductions = totalAmazonFees + totalCostPerUnit;
  const netProfit = p.sellingPrice - totalDeductions;
  const profitMargin = p.sellingPrice > 0 ? (netProfit / p.sellingPrice) * 100 : 0;
  const roi = totalCostPerUnit > 0 ? (netProfit / totalCostPerUnit) * 100 : 0;
  const effectiveRate = getEffectiveReferralRate(p.category);
  const breakEvenPrice = effectiveRate < 1 ? (totalCostPerUnit / (1 - effectiveRate)) + fbaFee + storageFeePerUnit : 0;
  const breakEvenUnits = netProfit > 0 ? Math.ceil((storageFeePerUnit * p.monthlyVolume) / netProfit) : 0;
  const targetROI = p.targetROI || 50;
  const suggestedPrice = effectiveRate < 1 ? (totalCostPerUnit * (1 + targetROI / 100) + fbaFee + storageFeePerUnit) / (1 - effectiveRate) : 0;
  const profitabilityBadge = getBadge(netProfit, profitMargin);

  return {
    ...p,
    referralFee, fbaFee, storageFeePerUnit, totalAmazonFees,
    totalCostPerUnit, totalDeductions, netProfit, profitMargin,
    roi, breakEvenPrice, breakEvenUnits, suggestedPrice, profitabilityBadge
  };
}

  document.getElementById('op-status').textContent = count > 0 ? (products[products.length - 1].name || 'ready') : 'ready';

  let avgMargin = '--', avgRoi = '--';
  if (count > 0) {
    const sumMargin = products.reduce((s, p) => s + (p.profitMargin || 0), 0);
    const sumRoi = products.reduce((s, p) => s + (p.roi || 0), 0);
    avgMargin = fmtPct(sumMargin / count) + '%';
    avgRoi = fmtPct(sumRoi / count) + '%';
  }
  document.getElementById('rt-margin').textContent = avgMargin;
  document.getElementById('rt-roi').textContent = avgRoi;
  document.getElementById('tele-margin').textContent = avgMargin;

  // Runtime bars
  const healthyStrong = products.filter(p => p.profitabilityBadge === 'healthy' || p.profitabilityBadge === 'strong').length;
  const bar1Pct = count > 0 ? (healthyStrong / count * 100) : 0;
  const bar2Pct = count > 0 ? ((count - healthyStrong) / count * 100) : 0;
  document.getElementById('rt-bar1').style.width = bar1Pct + '%';
  document.getElementById('rt-bar2').style.width = bar2Pct + '%';
}

function saveProduct() {
  const name = document.getElementById('c-name').value.trim();
  if (!name) { showToast('Product name is required', 'error'); return; }
  const p = calculateProduct({
    id: uid(), name, asin: document.getElementById('c-asin').value.trim(),
    category: document.getElementById('c-category').value,
    sizeTier: document.getElementById('c-sizeTier').value,
    weight: parseFloat(document.getElementById('c-weight').value) || 0,
    productCost: parseFloat(document.getElementById('c-productCost').value) || 0,
    prepCost: parseFloat(document.getElementById('c-prepCost').value) || 0,
    inboundShipping: parseFloat(document.getElementById('c-inboundShipping').value) || 0,
    sellingPrice: parseFloat(document.getElementById('c-sellingPrice').value) || 0,

}

function setPill(id, val, suffix, isPct) {
  const el = document.getElementById(id);
  const absVal = Math.abs(val);
  if (val > 0.01) { el.className = 'pill up'; el.textContent = '+' + fmtPct(absVal) + suffix; }

function sortProducts(col) {
  if (sortCol === col) sortDir = sortDir === 'asc' ? 'desc' : 'asc';
  else { sortCol = col; sortDir = 'asc'; }
  renderProductsTable();
}

function getFilteredSortedProducts() {
  let list = [...products];
  // Badge filter
  if (productFilterBadge !== 'all') {
    list = list.filter(p => p.profitabilityBadge === productFilterBadge);
  }
  // Size filter
  if (productFilterSize !== 'all') {
    list = list.filter(p => p.sizeTier === productFilterSize);
  }
  // Search
  const search = document.getElementById('product-search')?.value?.trim().toLowerCase() || '';
  if (search) {
    list = list.filter(p => (p.name || '').toLowerCase().includes(search) || (p.asin || '').toLowerCase().includes(search));
  }
  // Sort
  list.sort((a, b) => {
    let av = a[sortCol], bv = b[sortCol];
    if (typeof av === 'string') { av = av.toLowerCase(); bv = (bv || '').toLowerCase(); }
    else { av = av || 0; bv = bv || 0; }
    if (av < bv) return sortDir === 'asc' ? -1 : 1;
    if (av > bv) return sortDir === 'asc' ? 1 : -1;
    return 0;
  });
  return list;
}

function renderProductsTable() {
  const tbody = document.getElementById('data-tbody');
  const list = getFilteredSortedProducts();
  document.getElementById('count-label').textContent = list.length + ' shown';
  // Update header sort indicators
  document.querySelectorAll('#view-data thead th').forEach(th => {
    th.classList.remove('sort-asc', 'sort-desc');
    const onclick = th.getAttribute('onclick');
    if (onclick) {
      const match = onclick.match(/sortProducts\('([^']+)'\)/);
      if (match && match[1] === sortCol) {
        th.classList.add(sortDir === 'asc' ? 'sort-asc' : 'sort-desc');
      }
    }
  });
  if (list.length === 0) {
    tbody.innerHTML = '<tr><td colspan="14"><div class="empty-state"><div class="empty-icon">&#8709;</div><h3>NO PRODUCTS</h3><p>No products match your filters.</p></div></td></tr>';
    document.getElementById('data-pagination').innerHTML = '';
    return;
  }
  tbody.innerHTML = list.map(p => {
    const badgeClass = p.profitabilityBadge || 'thin';
    const dotClass = badgeClass === 'loss' ? 'red' : badgeClass === 'thin' ? 'amber' : badgeClass === 'healthy' ? 'green' : 'gray';
    const checked = selectedIds.has(p.id) ? 'checked' : '';
    return '<tr>' +
      '<td><input type="checkbox" class="tbl-check" ' + checked + ' onchange="toggleSelect(\'' + p.id + '\')"></td>' +
      '<td><div class="id"><span class="dot ' + dotClass + '"></span><b>' + escapeHtml(p.name || 'Untitled') + '</b></div></td>' +
      '<td>' + escapeHtml(p.asin || '--') + '</td>' +
      '<td>' + escapeHtml(p.category) + '</td>' +
      '<td>' + escapeHtml(p.sizeTier || '--') + '</td>' +
      '<td class="num">$' + fmt$(p.totalCostPerUnit || 0) + '</td>' +
      '<td class="num">$' + fmt$(p.sellingPrice || 0) + '</td>' +
      '<td class="num">$' + fmt$(p.totalAmazonFees || 0) + '</td>' +
      '<td class="num" style="color:' + (p.netProfit >= 0 ? 'var(--signal)' : 'var(--danger)') + '">$' + fmt$(p.netProfit || 0) + '</td>' +
      '<td class="num">' + fmtPct(p.profitMargin || 0) + '%</td>' +
      '<td class="num">' + fmtPct(p.roi || 0) + '%</td>' +
      '<td><span class="badge ' + badgeClass + '">' + badgeClass + '</span></td>' +
      '<td>' + (p.savedAt ? new Date(p.savedAt).toLocaleDateString() : '--') + '</td>' +
      '<td><div class="row-actions">' +
        '<button class="btn" type="button" onclick="loadProductToCalculator(\'' + p.id + '\')">load</button>' +
        '<button class="btn" type="button" onclick="duplicateProduct(\'' + p.id + '\')">dup</button>' +
        '<button class="btn danger" type="button" onclick="confirmDeleteProduct(\'' + p.id + '\')">del</button>' +
      '</div></td></tr>';
  }).join('');
  document.getElementById('data-pagination').innerHTML = '';
}

function updateCompare() {
  const colA = getCompareColumn('a');
  const colB = getCompareColumn('b');
  const resA = calculateProduct(colA);
  const resB = calculateProduct(colB);
  updateCompareCol('a', resA);
  updateCompareCol('b', resB);
  // Winner banner
  const banner = document.getElementById('winner-banner');
  const aNet = resA.netProfit, bNet = resB.netProfit;
  const aRoi = resA.roi, bRoi = resB.roi;
  if (aNet === 0 && bNet === 0 && aRoi === 0 && bRoi === 0) {
    banner.style.display = 'none';
  } else {
    banner.style.display = 'flex';
    let txt = '';
    if (aNet > bNet && aRoi > bRoi) txt = 'Winner: Product A — Higher net profit and ROI';
    else if (bNet > aNet && bRoi > aRoi) txt = 'Winner: Product B — Higher net profit and ROI';
    else if (aNet > bNet) txt = 'Winner: Product A — Higher net profit. Product B has better ROI';
    else if (bNet > aNet) txt = 'Winner: Product B — Higher net profit. Product A has better ROI';
    else if (aRoi > bRoi) txt = 'Product A has better ROI. Net profit is equal';
    else if (bRoi > aRoi) txt = 'Product B has better ROI. Net profit is equal';
    else txt = 'Both products are equivalent';
    document.getElementById('winner-text').textContent = txt;
  }
  saveCompareState();
}

function renderAnalytics() {
  const container = document.getElementById('chart-profit');
  if (products.length === 0) {
    container.innerHTML = '<div class="empty-state" style="border:none;padding:20px"><div class="empty-icon">&#8709;</div><h3>NO DATA</h3><p>Save products to see analytics.</p></div>';
    document.getElementById('sum-count').textContent = '0';
    document.getElementById('sum-margin').textContent = '--';
    document.getElementById('sum-roi').textContent = '--';
    document.getElementById('sum-best').textContent = '--';
    document.getElementById('sum-category').textContent = '--';
    ['hist-loss-bar','hist-thin-bar','hist-healthy-bar','hist-strong-bar'].forEach(id => document.getElementById(id).style.height = '0%');
    ['hist-loss-count','hist-thin-count','hist-healthy-count','hist-strong-count'].forEach(id => document.getElementById(id).textContent = '0');
    return;
  }
  // Summary stats
  const count = products.length;
  const avgMargin = products.reduce((s, p) => s + (p.profitMargin || 0), 0) / count;
  const avgRoi = products.reduce((s, p) => s + (p.roi || 0), 0) / count;
  const best = products.reduce((b, p) => ((p.netProfit || 0) > (b.netProfit || 0) ? p : b), products[0]);
  const catCounts = {};
  products.forEach(p => { catCounts[p.category] = (catCounts[p.category] || 0) + 1; });
  const topCat = Object.entries(catCounts).sort((a, b) => b[1] - a[1])[0];
  document.getElementById('sum-count').textContent = count;
  document.getElementById('sum-margin').textContent = fmtPct(avgMargin) + '%';
  document.getElementById('sum-roi').textContent = fmtPct(avgRoi) + '%';
  document.getElementById('sum-best').textContent = escapeHtml((best.name || 'Untitled').substring(0, 20));
  document.getElementById('sum-category').textContent = topCat ? escapeHtml(topCat[0]) : '--';

  // Profit bar chart (top 10 by net profit)
  const top10 = [...products].sort((a, b) => (b.netProfit || 0) - (a.netProfit || 0)).slice(0, 10);
  const maxNet = Math.max(...top10.map(p => Math.abs(p.netProfit || 0)), 1);
  container.innerHTML = top10.map(p => {
    const h = (Math.abs(p.netProfit || 0) / maxNet * 100);
    const cls = p.netProfit >= 0 ? 'healthy' : 'loss';
    return '<div class="bar-chart-group"><div class="bar-chart-val">$' + fmt$(p.netProfit) + '</div><div class="bar-chart-bar ' + cls + '" style="height:' + h + '%"></div><div class="bar-chart-label">' + escapeHtml((p.name || '').substring(0, 12)) + '</div></div>';
  }).join('');

  // Margin distribution histogram
  const dist = { loss: 0, thin: 0, healthy: 0, strong: 0 };
  products.forEach(p => { dist[p.profitabilityBadge || 'thin'] = (dist[p.profitabilityBadge || 'thin'] || 0) + 1; });
  const maxDist = Math.max(dist.loss, dist.thin, dist.healthy, dist.strong, 1);
  document.getElementById('hist-loss-count').textContent = dist.loss;
  document.getElementById('hist-loss-bar').style.height = (dist.loss / maxDist * 100) + '%';
  document.getElementById('hist-thin-count').textContent = dist.thin;
  document.getElementById('hist-thin-bar').style.height = (dist.thin / maxDist * 100) + '%';
  document.getElementById('hist-healthy-count').textContent = dist.healthy;
  document.getElementById('hist-healthy-bar').style.height = (dist.healthy / maxDist * 100) + '%';
  document.getElementById('hist-strong-count').textContent = dist.strong;
  document.getElementById('hist-strong-bar').style.height = (dist.strong / maxDist * 100) + '%';
}

    accentHue: 32
  });
  applyAccent(currentSettings.accentHue);
  refreshDropdowns();

  if (catEl && currentSettings.defaultCategory) {
    catEl.value = currentSettings.defaultCategory;
  }
  const tierEl = document.getElementById('c-sizeTier');
  if (tierEl && currentSettings.defaultSizeTier) tierEl.value = currentSettings.defaultSizeTier;
  if (currentSettings.defaultStorageMonth === 'q4') {
    document.getElementById('c-storageMonth').checked = true;
  }

    document.getElementById('c-targetROI').value = currentSettings.defaultTargetROI;
    document.getElementById('c-targetROI-val').textContent = currentSettings.defaultTargetROI + '%';
  }
  // Settings UI
  const sc = document.getElementById('s-defaultCategory');
  if (sc && currentSettings.defaultCategory) sc.value = currentSettings.defaultCategory;
  const st = document.getElementById('s-defaultSizeTier');
  if (st && currentSettings.defaultSizeTier) st.value = currentSettings.defaultSizeTier;

  document.getElementById('s-accentHue-val').textContent = currentSettings.accentHue || 32;
  updateCalculator();
}

function refreshDropdowns() {
  const selects = ['c-category', 'a-category', 'b-category', 's-defaultCategory'];
  selects.forEach(selId => {
    const sel = document.getElementById(selId);
    if (!sel) return;
    const oldVal = sel.value;
    sel.innerHTML = CATEGORIES.map(c => '<option value="' + escapeHtml(c.name) + '">' + escapeHtml(c.name) + '</option>').join('');
    if (oldVal) sel.value = oldVal;
    else if (selId === 's-defaultCategory' && currentSettings.defaultCategory) sel.value = currentSettings.defaultCategory;
    else if (currentSettings.defaultCategory) sel.value = currentSettings.defaultCategory;
  });
}

  openModal('Clear All Data', '<p>This will delete ALL saved products and reset all settings. This cannot be undone.</p>',

  products = [];
  selectedIds.clear();
  saveProducts();
  LS.del('settings');

  loadSettings();

function exportJSON() {
  const data = {
    products: products,
    settings: currentSettings,
    exportedAt: new Date().toISOString()
  };
  const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'fba-calc-backup-' + new Date().toISOString().split('T')[0] + '.json';
  a.click();
  URL.revokeObjectURL(url);
  showToast('JSON backup exported', 'success');
}

function importJSON(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = function(e) {
    try {
      const data = JSON.parse(e.target.result);
      if (data.products && Array.isArray(data.products)) {
        products = data.products;
        saveProducts();
        renderProductsTable();
      }
      if (data.settings) {
        currentSettings = data.settings;
        LS.set('settings', currentSettings);
        loadSettings();
      }
      showToast('Data imported successfully', 'success');
    } catch(err) {
      showToast('Invalid JSON file', 'error');
    }
    input.value = '';
  };
  reader.readAsText(file);
}

function exportToCSV() {
  if (products.length === 0) { showToast('No products to export', 'info'); return; }
  const headers = ['Name','ASIN','Category','Size Tier','Weight (lbs)','Product Cost','Prep Cost','Inbound Shipping',
    'Selling Price','Monthly Volume','Storage Month','Target ROI','Storage Volume',
    'Referral Fee','FBA Fee','Storage Fee','Total Amazon Fees','Total Cost Per Unit',
    'Total Deductions','Net Profit','Profit Margin %','ROI %','Break Even Price','Break Even Units',
    'Suggested Price','Badge','Saved At'];
  const rows = products.map(p => [
    p.name, p.asin || '', p.category, p.sizeTier, p.weight, p.productCost, p.prepCost, p.inboundShipping,
    p.sellingPrice, p.monthlyVolume, p.storageMonth, p.targetROI, p.storageVolumeEstimate || 0.1,
    fmt$(p.referralFee), fmt$(p.fbaFee), fmt$(p.storageFeePerUnit), fmt$(p.totalAmazonFees), fmt$(p.totalCostPerUnit),
    fmt$(p.totalDeductions), fmt$(p.netProfit), fmtPct(p.profitMargin), fmtPct(p.roi), fmt$(p.breakEvenPrice), p.breakEvenUnits || 0,
    fmt$(p.suggestedPrice), p.profitabilityBadge, p.savedAt || ''
  ]);
  const csv = [headers.join(','), ...rows.map(r => r.map(v => '"' + String(v).replace(/"/g, '""') + '"').join(','))].join('\n');
  const blob = new Blob([csv], { type: 'text/csv' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'fba-products-' + new Date().toISOString().split('T')[0] + '.csv';
  a.click();
  URL.revokeObjectURL(url);
  showToast('Exported ' + products.length + ' products', 'success');
}

function renderHelpContent() {
  const catDiv = document.getElementById('help-categories');
  if (catDiv && !catDiv.dataset.rendered) {
    catDiv.innerHTML = CATEGORIES.map(c => {
      let rateStr = (c.rate * 100).toFixed(0) + '%';
      if (c.tiered) rateStr = (c.rate * 100).toFixed(0) + '% up to $' + c.tier1Threshold + ', ' + (c.tier2Rate * 100).toFixed(0) + '% above';
      if (c.minFee > 0) rateStr += ' (min $' + fmt$(c.minFee) + ')';
      return '<div class="fee-row"><b>' + escapeHtml(c.name) + '</b><span>' + rateStr + '</span></div>';
    }).join('');
    catDiv.dataset.rendered = '1';
  }
  const tierDiv = document.getElementById('help-size-tiers');
  if (tierDiv && !tierDiv.dataset.rendered) {
    let html = '';
    for (const [tierName, brackets] of Object.entries(FBA_FEES)) {
      html += '<div style="margin-bottom:12px"><strong style="font-size:11px;letter-spacing:.12em;text-transform:uppercase;color:var(--ink)">' + escapeHtml(tierName) + '</strong>';
      for (const b of brackets) {
        let label, fee;
        if (b.fee !== null) {
          label = 'up to ' + b.max + 'oz';
          fee = '$' + fmt$(b.fee);
        } else {
          label = b.max === Infinity ? b.above + 'lb+' : b.above + '-' + Math.round(b.max / 16) + 'lb';
          fee = '$' + fmt$(b.base) + ' + $' + fmt$(b.perLb) + '/lb above ' + b.above + 'lb';
        }
        html += '<div class="fee-row"><span>' + label + '</span><span>' + fee + '</span></div>';
      }
      html += '</div>';
    }
    tierDiv.innerHTML = html;
    tierDiv.dataset.rendered = '1';
  }
}
